jatizso_missions_3 = {
	slot = 3
	generic = no
	ai = yes
	potential = {
		tag = JTZ
		NOT = { map_setup = map_setup_random }
	}
	has_country_shield = yes
	
	JTZ_northward_expansion = {
		icon = mission_alliances
		required_missions = { JTZ_grow_the_economy JTZ_exploit_the_mines }
		position = 2
		trigger = {
			OR = {
				owns_core_province = 387
				AND = {
				treasury = 250
				adm_power = 50
				387 = {
					is_empty = yes
					}
				}
			}
		}
		effect = {
			if = {
				limit = {
					owns_core_province = 387
					}
				add_adm_power = 50
			}
			else = {
				add_treasury = -250
				add_adm_power = -50
				387 = {
					add_core = ROOT
					cede_province = ROOT
					change_trade_goods = basic_ores
					}
			}
		}
	}
}
