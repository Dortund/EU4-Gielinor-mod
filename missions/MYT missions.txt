morytania_missions_2 = {
	slot = 2
	generic = no
	ai = yes
	potential = {
		tag = MYT
		NOT = { map_setup = map_setup_random }
	}
	has_country_shield = yes
	
	MYT_reinforce_the_coast = {
		icon = mission_conquer_50_development
		required_missions = { }
		trigger = {
			calc_true_if = {
				all_owned_province = {
					has_port = yes
					OR = {
						has_building = coastal_defence
						has_building = naval_battery
						}
					}
				amount = 4
			}
		}
		effect = {
			add_adm_power = 100
		}
	}
	MYT_pirate_accommodation = {
		icon = mission_conquer_50_development
		required_missions = { MYT_reinforce_the_coast MYT_strengthen_vassal_ties }
		trigger = {
			OR = {
				AND = {
					reverse_has_opinion = {
						who = MSH
						value = 100
					}
					treasury = 250
					}
				AND = {
					owns_core_province = 142
					treasury = 100
					}
			}
		}
		effect = {
			add_country_modifier = {
				name = "MYT_pirate_mercenaries"
				duration = 3650
			}
			if = {
				limit = { owns_core_province = 142 }
				add_treasury = -100
			}
			else = {
				add_treasury = -250
					MSH = {
						country_event = {
						id = flavor_msh.1
						days = 1
						}
					}
			}
		}
	}
}
