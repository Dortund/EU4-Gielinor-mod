		icon = mission_soul_altar
		position = 3
		required_missions = { MPS_expand_inland }
		trigger = {
			owns_core_province = 858
			has_institution = runecrafting
			OR = {
				owns_core_province = 96

			}
		}
		effect = {
			if = {
				limit = {
					owns_core_province = 96

					add_base_production = 1
					add_base_manpower = 1
				}
			}
			else = {
				100 = {
					add_base_tax = -1
					add_base_production = -1
					add_base_manpower = -1
				}
				96 = {
					add_core = ROOT
					cede_province = ROOT				
					add_base_tax = 1
					add_base_production = 1
					add_base_manpower = 1
					hidden_effect = {
						add_institution_embracement = {
							which = reconstruction
							value = 100
						}
						add_institution_embracement = {
							which = architecture
							value = 100
						}
						add_institution_embracement = {
							which = runecrafting
							value = 100
						}
					}
				}
			}
		}
	}
	MPS_grand_library_refurbishment = {
		icon = mission_grand_library_refurbishment

		trigger = {
