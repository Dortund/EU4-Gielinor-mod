neitiznot_missions_3 = {
	slot = 3
	generic = no
	ai = yes
	potential = {
		tag = NZS
		NOT = { map_setup = map_setup_random }
	}
	has_country_shield = yes
	
	NZS_follow_the_arctic_pine = {
		icon = mission_alliances
		required_missions = { NZS_amenities_of_neitiznot NZS_yak_farming }
		position = 2
		trigger = {
			OR = {
				owns_core_province = 388
				AND = {
				treasury = 250
				adm_power = 50
				388 = {
					is_empty = yes
					}
				}
			}
		}
		effect = {
			if = {
				limit = {
					owns_core_province = 388
					}
				add_adm_power = 50
				388 = {
					change_trade_goods = runite
				}
			}
			else = {
				add_treasury = -250
				add_adm_power = -50
				388 = {
					add_core = ROOT
					cede_province = ROOT
					change_trade_goods = runite
				}
			}
		}
	}
}
