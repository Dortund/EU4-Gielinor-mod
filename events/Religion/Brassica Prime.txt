#Grow new cabbages
country_event = {
	id = brassicaprime_events.13
	title = brassicaprime_events.13.t #Grow the Holy Vegetable
	desc = brassicaprime_events.13.d #Brassican farmers of our realm have demanded to be allowed to grow the most sacred of vegetables, and our most exalted Cabbagemancers agree. Nonetheless, the decision falls to you.
	picture = BRASSICAPRIME_CABBAGE_eventPicture
	
	is_triggered_only = yes

	trigger = {
		religion = brassicaprime
		any_owned_province = {
			trade_goods = grain
			religion = brassicaprime
			NOT = { has_province_modifier = brassica_destroyed_cabbages }
			NOT = { has_province_modifier = brassica_refused_cabbages }
		}
	}
	
	mean_time_to_happen = {
		months = 36
		modifier = {
			factor = 1.5
			cabbage = 1
		}
		modifier = {
			factor = 1.5
			cabbage = 2
		}
		modifier = {
			factor = 1.5
			cabbage = 3
		}
		modifier = {
			factor = 1.5
			cabbage = 4
		}
	}
	
	option = {
		name = brassicaprime_events.13.a #Let them grow cabbages!
		
		random_owned_province = {
			limit = {
				trade_goods = grain
				religion = brassicaprime
				NOT = { has_province_modifier = brassica_destroyed_cabbages }
				NOT = { has_province_modifier = brassica_refused_cabbages }
			}
		 	change_trade_goods = cabbage
		 	add_province_modifier = {
				name = brassica_sacred_cabbages
				duration = 7300
			}
		}
		add_estate_church_loyalty_effect = yes
		ai_chance = {
			factor = 1
		}
	}
	option = {
		name = brassicaprime_events.13.b #Absence makes the heart grow fonder - for all things.
		random_owned_province = {
			limit = {
				trade_goods = grain
				religion = brassicaprime
				NOT = { has_province_modifier = brassica_destroyed_cabbages }
				NOT = { has_province_modifier = brassica_refused_cabbages }
			}
		 	add_province_modifier = {
				name = brassica_refused_cabbages
				duration = 7300
			}
		}
		reduce_estate_church_loyalty_effect = yes
		ai_chance = {
			factor = 0
		}
	}
}
