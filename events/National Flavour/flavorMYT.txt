#Vampyric Population Boom
country_event = {
	id = flavor_myt.2
	title = flavor_myt.2.t
	desc = flavor_myt.2.d #The vampyric population has surged in recent years, and the more vampyres we have, the easier it is to create more. There are concerns that converting so many humans to vampyrism will deplete our blood stores, but on the other hand, fewer potentially rebellious humans to worry about.
	picture = VAMPYRIC_eventPicture
	is_triggered_only = yes
	
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					NOT = { is_in_capital_area = yes }
					OR = {
						culture_group = misthalin_group
						culture_group = wushanko_group
					}
				}
				save_event_target_as = local_vampyric_conversion1
			}
			random_owned_province = {
				limit = {
					NOT = { is_in_capital_area = yes }
					OR = {
						culture_group = misthalin_group
						culture_group = wushanko_group
					}
				}
				save_event_target_as = local_vampyric_conversion2
			}
			random_owned_province = {
				limit = {
					NOT = { is_in_capital_area = yes }
					OR = {
						culture_group = misthalin_group
						culture_group = wushanko_group
					}
				}
				save_event_target_as = local_vampyric_conversion3
			}
		}
	}
	
	option = {
		name = flavor_myt.2.a #We need to maximise the numbers of these humanborn vampyres!
		event_target:local_vampyric_conversion1 = {
			change_culture = humanborn_vampyre
		}
		event_target:local_vampyric_conversion2 = {
			change_culture = humanborn_vampyre
		}
		event_target:local_vampyric_conversion3 = {
			change_culture = humanborn_vampyre
		}
		ai_chance = { factor = 100 }
	}
	option = {
		name = flavor_myt.2.b #Eww, human-born vampyres? No thanks.
		if = {
			limit = {
				has_government_mechanic = blood_power_mechanic
			}
			add_government_power = {
				mechanic_type = blood_power_mechanic
				power_type = blood_power_power_1
				value = 5
			}
		}
		else = {
			add_adm_power = 50
		}
		ai_chance = { factor = 0 }
	}
}

#Damien's Plan
country_event = {
	id = flavor_myt.3
	title = flavor_myt.3.t
	desc = flavor_myt.3.d #A vampyre noble by the name of Damien Leucurte has a plan to meet blood tithe shortages: Recreate the Sleeper Plague that once afflicted the town of Slepe and tithe the sleeping inhabitants. Victims of the plague can be tithed more heavily than those who remain awake.
	picture = VAMPYRIC_eventPicture
	fire_only_once = yes
	
	trigger = {
		tag = MYT
		culture_group = vampyre_group
		OR = {
			absolutism = 35
			innovativeness = 15
		}
		OR = {
			NOT = { has_reform = vampyre_kingdom }
			has_disaster = excessive_tithe
			NOT = {
				has_government_power = {
					mechanic_type = blood_power_mechanic
					power_type = blood_power_power_1
					value = 30
				}
			}
		}
		slepe_area = {
			type = all
			owned_by = ROOT
			is_core = ROOT
		}
	}

	mean_time_to_happen = {
		months = 240
	}
	
	option = {
		name = flavor_myt.3.a #A wonderful idea! Let us back this plan whole-heartedly!
		custom_tooltip = MYT_nightmare_risk_tooltip
		if = {
			limit = {
				has_reform = vampyre_kingdom
			}
			add_country_modifier = {
				name = MYT_tithe_the_sleeping
				duration = 3650
			}
		}
		else = {
			add_country_modifier = {
				name = MYT_tithe_the_sleeping_alt
				duration = 3650
			}
		}
		slepe_area = {
			add_province_modifier = {
				name = MYT_terror_of_the_sleeper_plague
				duration = 3650
			}
		}
		hidden_effect = {
			set_country_flag = MYT_new_sleeper_plague_flag
		}
		ai_chance = { factor = 20 }
	}
	option = {
		name = flavor_myt.3.b #An impressive solution, but it is likely to draw unwanted attention.
		custom_tooltip = MYT_nightmare_risk_tooltip
		if = {
			limit = {
				has_reform = vampyre_kingdom
			}
			add_country_modifier = {
				name = MYT_tithe_some_of_the_sleeping
				duration = 3650
			}
		}
		else = {
			add_country_modifier = {
				name = MYT_tithe_some_of_the_sleeping_alt
				duration = 3650
			}
		}
		slepe_area = {
			add_province_modifier = {
				name = MYT_mild_terror_of_the_sleeper_plague
				duration = 3650
			}
		}
		hidden_effect = {
			set_country_flag = MYT_new_sleeper_plague_flag
		}
		ai_chance = { factor = 80 }
	}
	option = {
		name = flavor_myt.3.c #We cannot approve this horrific idea.
		add_prestige = 5
		ai_chance = { factor = 0 }
	}
}
