country_event = {
	id = flavor_cfs.1
	title = flavor_cfs.1.t #Werewolf Contact
	desc = flavor_cfs.1.d #A population of werewolves from Morytania have reached out to us seeking co-operation. They have strong warriors and conviction in the true faith, but would benefit from assistance expanding their reach in the region. Aiding the werewolves could pay off with a strong alliance down the road.
	
	picture = CANIFIS_WEREWOLF_eventPicture
	
	is_triggered_only = yes
	
	option = {
		name = flavor_cfs.1.a #We would not turn away true followers of the [Root.Religion.GetName] faith.
		add_opinion = {
			who = CFS
			modifier = cfs_allies_in_faith
		}
		add_historical_friend = CFS
		CFS = {
			add_historical_friend = ROOT
		}
		CFS = {
			country_event = { 
				id = flavor_cfs.2
				days = 5
			}
		}
		ai_chance = {
			factor = 10
			modifier = {
				factor = 0
				alliance_with = MYT
			}
		}
	}
	
	option = {
		name = flavor_cfs.1.b #Tell these dogs that their fight is not ours.
		add_opinion = {
			who = CFS
			modifier = cfs_rejected_support
		}
		CFS = {
			country_event = { 
				id = flavor_cfs.3
				days = 5
			}
		}
		ai_chance = {
			factor = 0
			modifier = {
				factor = 100
				alliance_with = MYT
			}
		}
	}
}

		}
	}

country_event = {
	id = flavor_cfs.5
	title = flavor_cfs.5.t #The Fate of Slepe
	desc = flavor_cfs.5.d #The village of Slepe was formed by humans who saw the folly of Saradomin's ways and instead joined Lord Zamorak. As fellow Zamorakians, they perhaps should be granted more privileges than the heretical vampyres - and they should repay the favour in turn.
	
	picture = MORYTANIA_SWAMP_eventPicture
	
	is_triggered_only = yes
	
	option = {
		name = flavor_cfs.5.a #Provide them with limited self-governance.
		create_march = SLP
		custom_tooltip = cfs_tooltip_support_of_slepe
		ai_chance = {
			factor = 10
		}
	}
	
	option = {
		name = flavor_cfs.5.b #Give them more equal status in our realm.
		trigger = {
			NOT = {	accepted_culture = slepe }
		}
		add_accepted_culture = slepe
		ai_chance = {
			factor = 0
		}
	}
	option = {
		name = flavor_cfs.5.c #Let's put this whole Slepe situation to rest.
		every_owned_province = {
			limit = {
				culture = slepe
			}
			add_province_modifier = {
				name = CFS_oppression_of_slepe
				duration = 3650
			}
		}
		ai_chance = {
			factor = 0
		}
	}
}
