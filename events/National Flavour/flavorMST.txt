country_event = {
	id = flavor_mst.2
	title = flavor_mst.2.t #Fate of the Narogoshuun
	desc = flavor_mst.2.d #We have expanded our land into a region with a heavy population of Narogoshuun goblins. Some within our realm see their large numbers as a potential future threat from within, and want them exiled from Misthalin. However, the only place we can realistically send them is into Asgarnia, which is likely to be faced with alarm all over the region. What should we do?
	picture = MISTHALIN_LUMBRIDGE_GOBLINS_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = MST
		calc_true_if = {
			any_owned_province = {
				culture = narogoshuun
			 	is_core = MST
				}
			amount = 2
		}
		current_age = age_of_mortals
	}

	mean_time_to_happen = {
		months = 3
	}
	
	option = {
		name = flavor_mst.2.a #Drive the Narogoshuun to Asgarnia!
		ai_chance = { factor = 100 }
		add_mil_power = -25
	 	every_owned_province = {
			limit = { culture = narogoshuun }
			change_culture = lummer
			change_religion = ROOT
			add_base_manpower = -1
			remove_core = NGU
		}
	 	random_country = {
			limit = {
				capital_scope = { region = asgarnia_region }
				NOT = {
					tag = ENT
					tag = CDR
				}
				num_of_provinces_in_states = 2
			}
			country_event = {
				id = flavor_mst.3 #Narogoshuun Refugees
				days = 1
			}
		}
		every_country = {
			limit = {
				capital_scope = { region = asgarnia_region }
				NOT = { tag = CDR }
				NOT = { tag = ENT }
			}
			add_trust = {
				who = MST
				value = -5
				mutual = yes
			}
		}
	}
	option = {
		name = flavor_mst.2.b #Let them stay.
		ai_chance = { factor = 0 }

			limit = { culture = narogoshuun }
			add_province_modifier = {
				name = "MST_goblin_tolerance"
				duration = 7300
				}
		}
	}
}
