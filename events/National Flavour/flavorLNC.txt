namespace = flavor_lnc
normal_or_historical_nations = yes

# Vampyres Want Our Blood!
country_event = {
	id = flavor_lnc.1
	title = flavor_lnc.1.t
	desc = flavor_lnc.1.d #We thought them myth, but Morytania turns out to be swarming with terrifying vampyres, who demand our people as blood tithes! A scientist within our realm has argued our Lordship has no chance of survival against them without allowing some concessions.
	picture = MORYTANIA_VYREWATCH_eventPicture
	
	fire_only_once = yes

	trigger = {
		tag = LNC
		exists = MYT
		MYT = {
			total_development = 150
		}
	 	has_discovered = 140
		OR = {
			AND = {
				war_with = MYT
				NOT = {
					war_score_against = {
						who = MYT
						value = -25
					}
				}
			}
			AND = {
				has_institution = runecrafting
				NOT = { total_development = 125 }
				NOT = { absolutism = 10 }
			}
		}
	}

	mean_time_to_happen = {
		months = 2
	}

	immediate = {
	}
	
	option = {
		name = flavor_lnc.1.a #We must comply.
		custom_tooltip = LNC_fenkenstrains_plan_tooltip
		if = {
			limit = {
				CFS = {
					is_subject_of = MYT
				}
			}
			every_owned_province = {
				limit = {
					NOT = { area = lordship_of_the_north_coast_area }
				}
				cede_province = CFS
			}
		}
		else = {
			every_owned_province = {
				limit = {
					NOT = { area = lordship_of_the_north_coast_area }
				}
				cede_province = CFS
			}
		}
		MYT = {
			country_event = { 
				id = flavor_myt.7
				days = 30
			}
		}
		ai_chance = { factor = 80 }
	}
	option = {
		name = flavor_lnc.1.b #We will resist the Vampyres no matter what it takes!
		ai_chance = { factor = 20 }
	}
}
